% Def: Paragraph break
\newcommand{\breakpara}{\leavevmode\par\vspace{0.3em}}

% Def:    Ruleblock template
% Arg #1: Title (e.g., Implication)
% Arg #2: Tree Contents (minipage, test trees)
% Arg #3: Comment
\newcommand{\ruleblock}[3]{
    \noindent\rule{\textwidth}{0.4pt}
    \vspace{0.3em}
    \noindent \textbf{#1} 
    \begin{center} \small #2 \end{center}
    \vspace{-0.5em}
    \noindent {\small \textit{#3}}
    \vspace{1.5em}
}
