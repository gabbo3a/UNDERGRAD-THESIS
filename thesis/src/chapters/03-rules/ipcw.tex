\ruleblock{Quantificatore Universale ($\forall$)}{
    \begin{minipage}{0.48\textwidth}
        \centering
        \begin{prooftree}
            \hypo{\Phi}
            \infer1[($\forall I$)]{\forall x^\sigma \Phi}
        \end{prooftree}
    \end{minipage}%
    \hfill
    \begin{minipage}{0.48\textwidth}
        \centering
        \begin{prooftree}
            \hypo{\forall x^\sigma \Phi}
            \hypo{t:\sigma}
            \infer2[($\forall E$)]{\Phi[t/x]}
        \end{prooftree}
    \end{minipage}
}{Condizione sulla variabile: nella regola di introduzione ($\forall I$), la variabile $x^\sigma$ non 
deve comparire libera in alcuna ipotesi attiva al momento della derivazione \cite{hindley1997}.}

\ruleblock{Quantificatore Esistenziale ($\exists$)}{
    \begin{minipage}{0.45\textwidth}
        \centering
        \begin{prooftree}
            \hypo{t:\sigma}
            \hypo{\Phi[t/x]}
            \infer2[($\exists I$)]{\exists x^\sigma \Phi}
        \end{prooftree}
    \end{minipage}%
    \hfill
    \begin{minipage}{0.51\textwidth}
        \centering
        \begin{prooftree}
            \hypo{\exists x^\sigma \Phi}
            \hypo{[x:\sigma, \Phi]^i}
            \infer1{\Psi}
            \infer2[($\exists E^i$)]{\Psi}
        \end{prooftree}
    \end{minipage}
}{Condizione sulla variabile: nella regola di eliminazione ($\exists E$), la variabile $x^\sigma$ non 
deve essere libera in $\Psi$ o in qualunque altra ipotesi attiva esterna al ramo di scaricamento $i$ 
\cite{hindley1997}.}
