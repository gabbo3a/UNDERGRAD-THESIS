\begin{description}[leftmargin=3.5em, style=nextline]
    \item[$\forall x \Phi(x)$] Una funzione che, dato un elemento $d$ del dominio, restituisce una prova della proprietà $\Phi(d)$.
    \item[$\exists x \Phi(x)$] Una coppia $\langle d, e \rangle$ dove $d$ è l'individuo scelto (testimone) ed $e$ è l'evidenza che soddisfa $\Phi(d)$.
\end{description}
