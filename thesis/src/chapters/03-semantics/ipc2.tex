\begin{description}[leftmargin=3.5em, style=nextline]
    \item[$\forall X \phi(X)$] Una costruzione generica che, per ogni possibile proposizione $A$, genera una prova di $\phi(A)$.
    \item[$\exists X \phi(X)$] Una coppia $\langle A, e \rangle$ composta da una specifica proposizione $A$ e dalla sua relativa prova $e$.
\end{description}
